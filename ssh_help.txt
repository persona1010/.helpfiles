#------------------------------------------------------------------------------
# ssh-keygen 
#------------------------------------------------------------------------------
ssh-keygen [opt] - утилита для создания ключей.
    opt:
    -t - устанавливает тип ключа.
    -C "str" - добавляет комментарий. По умолчанию вписывается имя пользователя
        и машины.

    ex:
    #ssh-keygen -o -t rsa -C "e@mail.here" - генерирует ключ для github.

# ssh 
#------------------------------------------------------------------------------
#ssh [opt] USER@IP_ADDR
    opt:
    -i <path_to_key> - использовать указанный ключ для подключения (указывается
    мия приватного ключа).
    -p <port> - использует указанный порт (по умолчанию 22).
